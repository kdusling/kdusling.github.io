\documentclass[tikz,border=7pt]{standalone}
\usepackage{tikz,pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{calc}
\usetikzlibrary{angles,quotes} % for pic
\usetikzlibrary{patterns,decorations.pathmorphing}
\usetikzlibrary{arrows}
\usetikzlibrary{arrows.meta}
\usetikzlibrary{calc} 
\tikzset{>=latex}

\begin{document}
\def\a{3}
\def\b{0.5}
\def\c{0.01}

\pgfmathdeclarefunction{myx}{1}{%
  \pgfmathparse{\a+#1)}%
}
\pgfmathdeclarefunction{myy}{1}{%
  \pgfmathparse{\b*#1 + \c*#1^4)}%
}

\pgfmathdeclarefunction{myvnorm}{1}{%
  \pgfmathparse{(1+(\b+4*\c*#1^3)^2)^(0.5)}%
}

\pgfmathdeclarefunction{myvx}{1}{%
  \pgfmathparse{1.0/myvnorm(#1)}%
}
\pgfmathdeclarefunction{myvy}{1}{%
  \pgfmathparse{(\b + 4*\c*#1^3)/myvnorm(#1)}%
}

\pgfmathdeclarefunction{myax}{1}{%
  \pgfmathparse{-1*(\b+4*\c*#1^3)/myvnorm(#1)}%
}
\pgfmathdeclarefunction{myay}{1}{%
\pgfmathparse{(1)/myvnorm(#1)}%
}

\begin{tikzpicture}[]
    \begin{axis}[
        xlabel={$x$},
        ylabel={$y$},
        %grid=major,
        axis lines=center,
        xmin=-1, xmax=10,
        ymin=-1, ymax=6,
        %title={Parametric Plot with Time Parameter}
        xticklabel=\empty, 
        yticklabel=\empty,
        tick style={draw=none}
    ]
    \addplot[variable=\t,
        parametric,
        domain=0:6, % Domain for the parameter 't'
        samples=100,    % Number of samples 
        black,           % Line color
        line width=1pt  % Line thickness
    ] ({myx(t)},{myy(t)});

    \newcommand\ta{1}
    \newcommand\tb{3}

    \node[draw, fill=white, circle, inner sep=1.5pt, label=above left:$O$] at (0,0) {};
    
    \node[draw, fill=green!40!blue, rectangle, inner sep=1.5pt, label={[text=black]right:$P$}]at ({myx(\ta)},{myy(\ta)}) {};

    \node[draw, fill=green!40!blue, rectangle, inner sep=1.5pt, label={[text=black]right:$P^\prime$}] at ({myx(\tb)},{myy(\tb)}) {};

    %I rescaled the velocity by 2 since this is just a cartoon
    \draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] ({myx(\ta)},{myy(\ta)}) --++ (axis direction cs:{2.0*myvx(\ta)},{2.0*myvy(\ta)}) node[right] {$\hat{e}_t$};

    \draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] ({myx(\ta)},{myy(\ta)}) --++ (axis direction cs:{2.0*myax(\ta)},{2.0*myay(\ta)}) node[left] {$\hat{e}_n$};

    \draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] ({myx(\tb)},{myy(\tb)}) --++ (axis direction cs:{2.0*myvx(\tb)},{2.0*myvy(\tb)}) node[right] {$\hat{e}_t^\prime$};

    \draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] ({myx(\tb)},{myy(\tb)}) --++ (axis direction cs:{2.0*myax(\tb)},{2.0*myay(\tb)}) node[left] {$\hat{e}_n^\prime$};
   
    \end{axis}
\end{tikzpicture}

\end{document}
