\documentclass[tikz,border=7pt]{standalone}
\usepackage{tikz,pgfplots}
\pgfplotsset{compat=1.18}
\usetikzlibrary{calc}
\usetikzlibrary{angles,quotes} % for pic
\usetikzlibrary{patterns,decorations.pathmorphing}
\usetikzlibrary{arrows}
\usetikzlibrary{arrows.meta}
\usetikzlibrary{calc} 
\tikzset{>=latex}

\begin{document}


\pgfmathdeclarefunction{myx}{1}{%
  \pgfmathparse{1 + #1 + 0.1*#1^2)}%
}
\pgfmathdeclarefunction{myy}{1}{%
  \pgfmathparse{0.5*#1 + 0.02*#1^4)}%
}
\pgfmathdeclarefunction{myvx}{1}{%
  \pgfmathparse{1.0+0.2*#1}%
}
\pgfmathdeclarefunction{myvy}{1}{%
  \pgfmathparse{0.5 + 0.08*#1^3)}%
}
\pgfmathdeclarefunction{myax}{1}{%
  \pgfmathparse{0.2}%
}
\pgfmathdeclarefunction{myay}{1}{%
  \pgfmathparse{0.24*#1^2)}%
}

\begin{tikzpicture}[]
    \begin{axis}[
        xlabel={$x$},
        ylabel={$y$},
        %grid=major,
        axis lines=center,
        xmin=-1, xmax=10,
        ymin=-1, ymax=10,
        %title={Parametric Plot with Time Parameter}
        xticklabel=\empty, 
        yticklabel=\empty,
        tick style={draw=none}
    ]
    \addplot[variable=\t,
        parametric,
        domain=0:6, % Domain for the parameter 't'
        samples=500,    % Number of samples 
        black,           % Line color
        line width=1pt  % Line thickness
    ] ({myx(t)},{myy(t)});

    \newcommand\ta{2.}
    \newcommand\tb{3.}

    \node[draw, fill=white, circle, inner sep=1.5pt, label=above left:$O$] at (0,0) {};
    
    \draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] (axis cs:{myx(\ta)},{myy(\ta)}) --++ (axis direction cs:{myvx(\ta)},{myvy(\ta)}) node[right] {$\vec{v}$};
    \draw[-{Latex[length=3mm,width=2mm]}, thick, red!40!black] (axis cs:{myx(\ta)},{myy(\ta)}) --++ (axis direction cs:{myax(\ta)},{myay(\ta)}) node[left] {$\vec{a}$};
	
	\draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] (axis cs:{myx(\tb)},{myy(\tb)}) --++ (axis direction cs:{myvx(\tb)},{myvy(\tb)}) node[right] {$\vec{v}$};
    \draw[-{Latex[length=3mm,width=2mm]}, thick, red!40!black] (axis cs:{myx(\tb)},{myy(\tb)}) --++ (axis direction cs:{myax(\tb)},{myay(\tb)}) node[left] {$\vec{a}$};
	
    \end{axis}
\end{tikzpicture}

\end{document}
