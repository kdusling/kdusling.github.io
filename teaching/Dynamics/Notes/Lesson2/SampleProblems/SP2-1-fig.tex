\documentclass[tikz]{standalone} 

\input{../../pulley.tex}

\begin{document}

\pgfmathdeclarefunction{myx}{1}{%
  \pgfmathparse{2+#1^3)}
}
\pgfmathdeclarefunction{myy}{1}{%
  \pgfmathparse{2*#1^2}
}
\pgfmathdeclarefunction{myvx}{1}{%
  \pgfmathparse{3*#1^2}
}
\pgfmathdeclarefunction{myvy}{1}{%
  \pgfmathparse{4*#1}
}
\pgfmathdeclarefunction{myax}{1}{%
  \pgfmathparse{6*#1}
}
\pgfmathdeclarefunction{myay}{1}{%
  \pgfmathparse{4.0}
}

\begin{tikzpicture}[]
    \begin{axis}[
        xlabel={$x$},
        ylabel={$y$},
        %grid=major,
        axis lines=center,
        xmin=-1, xmax=10.2,
        ymin=-1, ymax=10.2,
        %title={Parametric Plot with Time Parameter}
        %xticklabel=\empty, 
        %yticklabel=\empty,
        %tick style={draw=none}
    ]

    \newcommand\ta{0}
    \newcommand\tb{1.}
    \newcommand\tc{2.0}

    \node[draw, fill=white, circle, inner sep=1.5pt, label=below left:$O$] at (0,0) {};
    
    \node[draw, fill=green!40!blue, rectangle, inner sep=1.5pt]at ({myx(\ta)},{myy(\ta)}) {};
    \node[above, xshift=6mm] at ({myx(\ta)},{myy(\ta)}) {\footnotesize $t=0$};

    
    \node[draw, fill=green!40!blue, rectangle, inner sep=1.5pt]at ({myx(\tb)},{myy(\tb)}) {};
    \node[right, xshift=1mm] at({myx(\tb)},{myy(\tb)}) {\footnotesize $t=1$};

        
    \node[draw, fill=green!40!blue, rectangle, inner sep=1.5pt]at ({myx(\tc)},{myy(\tc)}) {};
    \node[above, xshift=-6mm] at({myx(\tc)},{myy(\tc)}) {\footnotesize $t=2$};

    \addplot[variable=\t,
        parametric,
        domain=0:1, % Domain for the parameter 't'
        samples=100,    % Number of samples 
       blue,           % Line color
       line width=3pt,  % Line thickness
       draw opacity=0.3
    ] ({myx(t)},{myy(t)});
    
    \addplot[variable=\t,
        parametric,
        domain=0:2, % Domain for the parameter 't'
        samples=100,    % Number of samples 
       black,           % Line color
       line width=1pt  % Line thickness
    ] ({myx(t)},{myy(t)});
    
    \draw[-{Latex[length=3mm,width=2mm]}, thick, green!40!black] ({myx(\tb)},{myy(\tb)}) --++ (axis direction cs:{myvx(\tb)},{myvy(\tb)}) node[above] {$\vec{v}$};
    
    \draw[-{Latex[length=3mm,width=2mm]}, thick, red] ({myx(\tb)},{myy(\tb)}) --++ (axis direction cs:{myax(\tb},{myay(\tb)}) node[right] {$\vec{a}$};
\end{axis}
\end{tikzpicture}


\end{document}
